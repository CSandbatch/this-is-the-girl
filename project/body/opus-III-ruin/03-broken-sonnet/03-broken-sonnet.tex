\PoemHead{Broken Sonnet}{(This Is Not A Sonnet)}

\begin{stanza}
\makebox[0.9\linewidth][l]{We\hspace*{4.4em}as one\hspace*{5.2em}}\\
\quad \makebox[0.9\linewidth][l]{Un\hspace*{2.2em},un\hspace*{3.0em}\hspace*{2.6em}delay;}\\
\quad \makebox[0.9\linewidth][l]{Each\hspace*{5.6em}a blade,\hspace*{3.8em}}\\
\makebox[0.9\linewidth][l]{Than\hspace*{5.0em}could\hspace*{4.6em}say.}\\
\end{stanza}

\begin{stanza}
\makebox[0.9\linewidth][l]{For what we\hspace*{4.2em}its due}\\
\quad \makebox[0.9\linewidth][l]{In\hspace*{6.4em}and will;}\\
\quad \makebox[0.9\linewidth][l]{Desire\hspace*{3.4em},and we\hspace*{4.8em}knew}\\
\makebox[0.9\linewidth][l]{\hspace*{4.6em}to law our\hspace*{5.2em}fulfill.}\\
\end{stanza}

\begin{stanza}
Yet had we suffered little, little sense\\
\quad Had reckoned out that adamant ire;\\
\quad So—our armors broken, wrenched, and tense\\
We pressed as one along the blade of fire.\\
\end{stanza}

\begin{stanza}
\quad Thus am I driven: \textit{Si fuerit amor\\
\quad Sicut amor tuus}—who loves as you adore?
\end{stanza}

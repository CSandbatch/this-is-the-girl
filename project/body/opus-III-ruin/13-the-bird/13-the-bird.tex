\PoemHead{The Bird}{}

\begin{stanza}
Exiled!\\
\end{stanza}

\begin{stanza}
Still I keep the Old Rite:\\
With spring’s first blaze\\
Loose I a bird\\
into the air.\\
\end{stanza}

\begin{stanza}
So small a mercy,\\
yet I feel it—\\
why curse God,\\
when I can grant\\
one creature\\
flight?\\
\end{stanza}

\begin{stanza}
I am\\
open now\\
to consolation\\
\end{stanza}

\begin{stanza}
God?\\
I will not rail\\
I have given\\
one living thing
\end{stanza}

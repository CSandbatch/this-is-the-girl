\PoemHead{III}{}

\begin{stanza}
They meet—ridge split by sudden light,\\
  The beast, a coiled burning living seam;\\
    The mountain frozen, stretched to white,\\
      Their gaze a scripture, rimmed in dream.\\
\end{stanza}

\begin{stanza}
He lifts the gun. The leopard stays—\\
  A god, a sister, none, or both—\\
    And in that stare, the old malaise\\
      Is loosed. He drops the gun, his oath\\
        Undone. Her eyes, a question laid\\
          On stone and snow, refuse the claim.\\
\end{stanza}

\begin{stanza}
Steel lowered, mountain heartbeat played\\
  Whispers private, a not-quite-human name.\\
\end{stanza}

\begin{stanza}
He leaves the wild, the wild remains,\\
  Two shadows lost in parting strains.
\end{stanza}

\CyclePartHead{I}

\begin{stanza}
Clarity speaks in doubt;\\
truthful men are scarce.\\
Where is the wise one?\\
My loyal,\\
brother who holds fast?\\
\end{stanza}

\begin{stanza}
Flicker in the dark, cast light\\
upon bodies without weight.\\
\end{stanza}

\begin{stanza}
Sadness settles in; I wait and wait.\\
My youth is gone,\\
hope burned down to ember.
\end{stanza}

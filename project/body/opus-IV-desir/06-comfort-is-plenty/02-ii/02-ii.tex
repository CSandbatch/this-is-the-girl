\CyclePartHead{II}

\begin{stanza}
Some stand near but drift far as fog.\\
Some hearts only open\\
when the weather turns.\\
\end{stanza}

\begin{stanza}
A true friend appears in trouble;\\
loyalty is thinnest\\
when comfort is plenty.\\
\end{stanza}

\begin{stanza}
Faithful ones fade,\\
longing aches under all this sky.
\end{stanza}

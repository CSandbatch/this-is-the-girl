\PoemHead{Spring Question}{}

\begin{stanza}
Fresh news ending a cold spring\\
\quad reviving parched lips\\
\end{stanza}

\begin{stanza}
Mountain for a house\\
(I have made)\\
animals for friends\\
\quad wind and sorrow\\
companions on the stone\\
\end{stanza}

\begin{stanza}
While joy endures\\
\quad no shade lasts.\\
\end{stanza}

\begin{stanza}
Swords will be praised\\
\quad by those who have bled.\\
\end{stanza}

\begin{stanza}
Mausoleum thistles\\
\quad flowering sands of time\\
\quad learn by walking\\
\quad endure a length of waiting\\
\end{stanza}

\begin{stanza}
Three days:\\
\quad hoping\\
\quad waiting\\
\quad leaving\\
\end{stanza}

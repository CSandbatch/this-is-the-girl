\PoemHead{Quartet For Two}{(Do You Want to Get the Girl?)}

\begin{stanza}
"Dude, four people are involved,"\\
says someone else.
\end{stanza}

\begin{stanza}
\textit{Standing plain: the real girl\\
sharpens the scene.\\
Suddenly attentive to real stakes,\\
it feels like Iâ€™m the one being hunted now.}
\end{stanza}

\begin{stanza}
Still the real girl,\\
standing where my thought curdles\\
itself into something solid.
\end{stanza}

\begin{stanza}
\textit{Brightens, suddenly more knowable,\\
I catch myself watching, slightly misgiven,\\
soften into innocence. Performative.\\
No one is doing harm.}
\end{stanza}

\begin{stanza}
My friend names them back:\\
``one who is only geometry,\\
the real one, the imagined one,''\\
as though speaking her reframes\\
the frame.
\end{stanza}

\begin{stanza}
\textit{Washed-out morning, spared from scrutiny.\\
Still real, I watch while wanting\\
to be without; wanting.}
\end{stanza}

\begin{stanza}
Real. I wish she constructed\\
me, real, and from a voice inside.
\end{stanza}

\begin{stanza}
\textit{Unadorned again. Returning, smaller,\\
almost nonimagined, spotted.\\
Angled. ``Stand a little closer, please.''}
\end{stanza}

\begin{stanza}
Her frame now looks back at me.\\
A quartet after all.
\end{stanza}

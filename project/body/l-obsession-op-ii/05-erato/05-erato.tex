\PoemHead{Erato}{}

\begin{stanza}
Chrysanthemum stroke.\\
Blossom / vanish / reappear.\\
\end{stanza}

\begin{stanza}
Bloodless \& baroque\\
Sidewinder Annunciation.\\
Never walking folding space.\\
\end{stanza}

\begin{stanza}
A slant arrival.\\
You quiet rival.\\
Intention curves sideways.\\
Shatters distance.\\
\end{stanza}

\begin{stanza}
One place, then another.\\
Layered lacquered silk lifting\\
buries you inside the fold.\\
\end{stanza}

\begin{stanza}
Never carried.\\
Never beckoned.\\
There then\\
There again.
\end{stanza}
